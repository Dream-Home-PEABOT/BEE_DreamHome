103class Document(BaseDocument, metaclass=TopLevelDocumentMetaclass): 

104 """The base class used for defining the structure and properties of 

105 collections of documents stored in MongoDB. Inherit from this class, and 

106 add fields as class attributes to define a document's structure. 

107 Individual documents may then be created by making instances of the 

108 :class:`~mongoengine.Document` subclass. 

109 

110 By default, the MongoDB collection used to store documents created using a 

111 :class:`~mongoengine.Document` subclass will be the name of the subclass 

112 converted to lowercase. A different collection may be specified by 

113 providing :attr:`collection` to the :attr:`meta` dictionary in the class 

114 definition. 

115 

116 A :class:`~mongoengine.Document` subclass may be itself subclassed, to 

117 create a specialised version of the document that will be stored in the 

118 same collection. To facilitate this behaviour a `_cls` 

119 field is added to documents (hidden though the MongoEngine interface). 

120 To enable this behaviourset :attr:`allow_inheritance` to ``True`` in the 

121 :attr:`meta` dictionary. 

122 

123 A :class:`~mongoengine.Document` may use a **Capped Collection** by 

124 specifying :attr:`max_documents` and :attr:`max_size` in the :attr:`meta` 

125 dictionary. :attr:`max_documents` is the maximum number of documents that 

126 is allowed to be stored in the collection, and :attr:`max_size` is the 

127 maximum size of the collection in bytes. :attr:`max_size` is rounded up 

128 to the next multiple of 256 by MongoDB internally and mongoengine before. 

129 Use also a multiple of 256 to avoid confusions. If :attr:`max_size` is not 

130 specified and :attr:`max_documents` is, :attr:`max_size` defaults to 

131 10485760 bytes (10MB). 

132 

133 Indexes may be created by specifying :attr:`indexes` in the :attr:`meta` 

134 dictionary. The value should be a list of field names or tuples of field 

135 names. Index direction may be specified by prefixing the field names with 

136 a **+** or **-** sign. 

137 

138 Automatic index creation can be disabled by specifying 

139 :attr:`auto_create_index` in the :attr:`meta` dictionary. If this is set to 

140 False then indexes will not be created by MongoEngine. This is useful in 

141 production systems where index creation is performed as part of a 

142 deployment system. 

143 

144 By default, _cls will be added to the start of every index (that 

145 doesn't contain a list) if allow_inheritance is True. This can be 

146 disabled by either setting cls to False on the specific index or 

147 by setting index_cls to False on the meta dictionary for the document. 

148 

149 By default, any extra attribute existing in stored data but not declared 

150 in your model will raise a :class:`~mongoengine.FieldDoesNotExist` error. 

151 This can be disabled by setting :attr:`strict` to ``False`` 

152 in the :attr:`meta` dictionary. 

153 """

709 def reload(self, *fields, **kwargs): 

710 """Reloads all attributes from the database. 

711 

712 :param fields: (optional) args list of fields to reload 

713 :param max_depth: (optional) depth of dereferencing to follow 

714 """ 

715 max_depth = 1 

716 if fields and isinstance(fields[0], int): 

717 max_depth = fields[0] 

718 fields = fields[1:] 

719 elif "max_depth" in kwargs: 

720 max_depth = kwargs["max_depth"] 

721 

722 if self.pk is None: 

723 raise self.DoesNotExist("Document does not exist") 

724 

725 obj = ( 

726 self._qs.read_preference(ReadPreference.PRIMARY) 

727 .filter(**self._object_key) 

728 .only(*fields) 

729 .limit(1) 

730 .select_related(max_depth=max_depth) 

731 ) 

732 

733 if obj: 

734 obj = obj[0] 

735 else: 

736 raise self.DoesNotExist("Document does not exist") 

737 for field in obj._data: 

738 if not fields or field in fields: 

739 try: 

740 setattr(self, field, self._reload(field, obj[field])) 

741 except (KeyError, AttributeError): 

742 try: 

743 # If field is a special field, e.g. items is stored as _reserved_items, 

744 # a KeyError is thrown. So try to retrieve the field from _data 

745 setattr(self, field, self._reload(field, obj._data.get(field))) 

746 except KeyError: 

747 # If field is removed from the database while the object 

748 # is in memory, a reload would cause a KeyError 

749 # i.e. obj.update(unset__field=1) followed by obj.reload() 

750 delattr(self, field) 

751 

752 self._changed_fields = ( 

753 list(set(self._changed_fields) - set(fields)) 

754 if fields 

755 else obj._changed_fields 

756 ) 

757 self._created = False 

758 return self

760 def _reload(self, key, value): 

761 """Used by :meth:`~mongoengine.Document.reload` to ensure the 

762 correct instance is linked to self. 

763 """ 

764 if isinstance(value, BaseDict): 

765 value = [(k, self._reload(k, v)) for k, v in value.items()] 

766 value = BaseDict(value, self, key) 

767 elif isinstance(value, EmbeddedDocumentList): 

768 value = [self._reload(key, v) for v in value] 

769 value = EmbeddedDocumentList(value, self, key) 

770 elif isinstance(value, BaseList): 

771 value = [self._reload(key, v) for v in value] 

772 value = BaseList(value, self, key) 

773 elif isinstance(value, (EmbeddedDocument, DynamicEmbeddedDocument)): 

774 value._instance = None 

775 value._changed_fields = [] 

776 return value

972 @classmethod 

973 def compare_indexes(cls): 

974 """Compares the indexes defined in MongoEngine with the ones 

975 existing in the database. Returns any missing/extra indexes. 

976 """ 

977 

978 required = cls.list_indexes() 

979 

980 existing = [] 

981 for info in cls._get_collection().index_information().values(): 

982 if "_fts" in info["key"][0]: 

983 index_type = info["key"][0][1] 

984 text_index_fields = info.get("weights").keys() 

985 existing.append([(key, index_type) for key in text_index_fields]) 

986 else:

989 extra = [index for index in existing if index not in required]

991 # if { _cls: 1 } is missing, make sure it's *really* necessary 

992 if [("_cls", 1)] in missing: 

993 cls_obsolete = False 

994 for index in existing: 

995 if includes_cls(index) and index not in extra: 

996 cls_obsolete = True 

997 break 

998 if cls_obsolete: 

999 missing.remove([("_cls", 1)]) 

1000 

1001 return {"missing": missing, "extra": extra}

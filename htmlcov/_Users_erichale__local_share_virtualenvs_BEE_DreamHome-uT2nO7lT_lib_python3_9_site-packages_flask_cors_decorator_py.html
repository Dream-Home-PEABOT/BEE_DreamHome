18def cross_origin(*args, **kwargs): 

19 """ 

20 This function is the decorator which is used to wrap a Flask route with. 

21 In the simplest case, simply use the default parameters to allow all 

22 origins in what is the most permissive configuration. If this method 

23 modifies state or performs authentication which may be brute-forced, you 

24 should add some degree of protection, such as Cross Site Forgery 

25 Request protection. 

26 

27 :param origins: 

28 The origin, or list of origins to allow requests from. 

29 The origin(s) may be regular expressions, case-sensitive strings, 

30 or else an asterisk 

31 

32 Default : '*' 

33 :type origins: list, string or regex 

34 

35 :param methods: 

36 The method or list of methods which the allowed origins are allowed to 

37 access for non-simple requests. 

38 

39 Default : [GET, HEAD, POST, OPTIONS, PUT, PATCH, DELETE] 

40 :type methods: list or string 

41 

42 :param expose_headers: 

43 The header or list which are safe to expose to the API of a CORS API 

44 specification. 

45 

46 Default : None 

47 :type expose_headers: list or string 

48 

49 :param allow_headers: 

50 The header or list of header field names which can be used when this 

51 resource is accessed by allowed origins. The header(s) may be regular 

52 expressions, case-sensitive strings, or else an asterisk. 

53 

54 Default : '*', allow all headers 

55 :type allow_headers: list, string or regex 

56 

57 :param supports_credentials: 

58 Allows users to make authenticated requests. If true, injects the 

59 `Access-Control-Allow-Credentials` header in responses. This allows 

60 cookies and credentials to be submitted across domains. 

61 

62 :note: This option cannot be used in conjuction with a '*' origin 

63 

64 Default : False 

65 :type supports_credentials: bool 

66 

67 :param max_age: 

68 The maximum time for which this CORS request maybe cached. This value 

69 is set as the `Access-Control-Max-Age` header. 

70 

71 Default : None 

72 :type max_age: timedelta, integer, string or None 

73 

74 :param send_wildcard: If True, and the origins parameter is `*`, a wildcard 

75 `Access-Control-Allow-Origin` header is sent, rather than the 

76 request's `Origin` header. 

77 

78 Default : False 

79 :type send_wildcard: bool 

80 

81 :param vary_header: 

82 If True, the header Vary: Origin will be returned as per the W3 

83 implementation guidelines. 

84 

85 Setting this header when the `Access-Control-Allow-Origin` is 

86 dynamically generated (e.g. when there is more than one allowed 

87 origin, and an Origin than '*' is returned) informs CDNs and other 

88 caches that the CORS headers are dynamic, and cannot be cached. 

89 

90 If False, the Vary header will never be injected or altered. 

91 

92 Default : True 

93 :type vary_header: bool 

94 

95 :param automatic_options: 

96 Only applies to the `cross_origin` decorator. If True, Flask-CORS will 

97 override Flask's default OPTIONS handling to return CORS headers for 

98 OPTIONS requests. 

99 

100 Default : True 

101 :type automatic_options: bool 

102 

103 """ 

104 _options = kwargs 

105 

106 def decorator(f): 

107 LOG.debug("Enabling %s for cross_origin using options:%s", f, _options) 

108 

109 # If True, intercept OPTIONS requests by modifying the view function, 

110 # replicating Flask's default behavior, and wrapping the response with 

111 # CORS headers. 

112 # 

113 # If f.provide_automatic_options is unset or True, Flask's route 

114 # decorator (which is actually wraps the function object we return) 

115 # intercepts OPTIONS handling, and requests will not have CORS headers 

116 if _options.get('automatic_options', True): 

117 f.required_methods = getattr(f, 'required_methods', set()) 

118 f.required_methods.add('OPTIONS') 

119 f.provide_automatic_options = False 

120 

121 def wrapped_function(*args, **kwargs): 

122 # Handle setting of Flask-Cors parameters 

123 options = get_cors_options(current_app, _options) 

124 

125 if options.get('automatic_options') and request.method == 'OPTIONS': 

126 resp = current_app.make_default_options_response() 

127 else: 

128 resp = make_response(f(*args, **kwargs)) 

129 

130 set_cors_headers(resp, options) 

131 setattr(resp, FLASK_CORS_EVALUATED, True) 

132 return resp 

133 

134 return update_wrapper(wrapped_function, f) 

135 return decorator
